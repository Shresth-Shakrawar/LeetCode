class Solution:    
    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:
        k = len(lists)
        print(f"{k=}")
        if k == 0:
            return None
        for i in range(1,k):
            lists[i] = self.mergeTwoLists(lists[i-1],lists[i])
        return lists[-1]
    def mergeTwoLists(self,l1,l2):
        dummy = ListNode(0,None)
        tail = dummy
        while l1 and l2:
            if l1.val < l2.val:
                tail.next = l1
                l1 = l1.next
                tail = tail.next
            else:
                tail.next = l2
                l2 = l2.next
                tail = tail.next
        while l1:
            tail.next = l1
            l1 = l1.next
            tail = tail.next
        while l2:
            tail.next = l2
            l2 = l2.next
            tail = tail.next
        return dummy.next
