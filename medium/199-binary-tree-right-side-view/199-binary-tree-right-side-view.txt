# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
from collections import deque
class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:
        res = []
        queue = deque([])
        if not root:
            return res
        queue.append(root)
        while queue:
            #print(f"{[x.val for x in queue]}")
            size = len(queue)
            while size > 1:
                node = queue.popleft()
                if node.left: 
                    queue.append(node.left)
                if node.right: 
                    queue.append(node.right)
                size -=1
            rightNode = queue.popleft()
            if rightNode.left:
                queue.append(rightNode.left)
            if rightNode.right:
                queue.append(rightNode.right)
            res.append(rightNode.val)
            
        return res
